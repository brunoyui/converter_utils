SELECT activity, COUNT(*) AS frequency, resource, CASE WHEN resource LIKE '%funcionário%' THEN 'Funcionário' WHEN resource LIKE '%supervisor%' THEN 'Supervisor' WHEN resource LIKE '%administração%' THEN 'Administração' WHEN resource LIKE '%pré-aprovador%' THEN 'Pré-aprovador' WHEN resource LIKE '%responsável pelo orçamento%' THEN 'Responsável pelo Orçamento' WHEN resource LIKE '%diretor%' THEN 'Diretor' ELSE 'Outro' END AS resource_group FROM event_log GROUP BY activity, resource, CASE WHEN resource LIKE '%funcionário%' THEN 'Funcionário' WHEN resource LIKE '%supervisor%' THEN 'Supervisor' WHEN resource LIKE '%administração%' THEN 'Administração' WHEN resource LIKE '%pré-aprovador%' THEN 'Pré-aprovador' WHEN resource LIKE '%responsável pelo orçamento%' THEN 'Responsável pelo Orçamento' WHEN resource LIKE '%diretor%' THEN 'Diretor' ELSE 'Outro' END ORDER BY frequency DESC, resource_group;
